from PIL import Image


img = Image.open("image.bmp")
width, height = img.size
part_size = width // 4
for row in range(1, 5):
    for col in range(1, 5):
        if row == 4 and col == 4:
            continue
        left = (col - 1) * part_size
        upper = (row - 1) * part_size
        right = col * part_size
        lower = row * part_size
        part = img.crop((left, upper, right, lower))
        part.save(f"image{row}{col}.bmp")
