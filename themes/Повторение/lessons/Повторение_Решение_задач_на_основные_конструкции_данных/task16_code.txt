import sys

classrooms = {}

for line in sys.stdin:
    line = line.rstrip('\n')

    if line == "":
        continue
    parts = line.rsplit(' ', 1)
    if len(parts) == 2 and parts[1].isdigit():
        subject = parts[0].strip()
        room_num = int(parts[1])
        if room_num not in classrooms:
            classrooms[room_num] = []
        if subject not in classrooms[room_num]:
            classrooms[room_num].append(subject)

sorted_rooms = sorted(classrooms.items())

for room_num, subjects in sorted_rooms:
    subjects_str = ", ".join(subjects)
    print(f"{room_num}: {subjects_str}")
